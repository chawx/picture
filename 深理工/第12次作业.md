布尔运算符包括逻辑与(`&&`)、逻辑或(`||`)和逻辑非(`!`)。这些运算符

```c
#include <stdio.h>

int main(int argc, char *argv[])
{
    int i = 0;

    if (argc == 1) {
        printf("You have no arguments. You suck.\n");
    } else if (argc > 1 && argc < 4) {
        printf("Here's your arguments:\n");

        for (i = 1; i < argc; i++) {
            printf("%s ", argv[i]);
        }
        printf("\n");
    } else {
        printf("You have too many arguments. You suck.\n");
    }

    // Testing additional cases
    if (argc == 2 && strcmp(argv[1], "openai") == 0) {
        printf("Welcome, OpenAI!\n");
    }

    // Using logical operators
    if (argc == 3 || (argc > 4 && argc <= 6)) {
        printf("You have a moderate number of arguments.\n");
    }

    // Using break statement to exit a loop
    for (i = 1; i < argc; i++) {
        //改动这里
        printf("%s ", argv[i]);
        if (i == 2) {
            printf("\nBreaking the loop early.\n");
            break;
        }
    }

    return 0;
}

```
